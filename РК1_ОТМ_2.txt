data = pd.read_csv(r'C:\Users\Artem\Downloads\toy_dataset.csv', sep=",")

print(data.head())

#Проведём Масштабирование 1 признака, а именно Age. Его значения находятся между 25 и 65, поэтому будем использовать MinMax масштабирование.

sc1 = MinMaxScaler()
sc1_data = sc1.fit_transform(data[['Age']])

plt.hist(data['Age'], 50)
plt.show()
plt.hist(sc1_data, 50)
plt.show()

#Проведём категоризацию по колонке City. Сначала проведём методом LabelEncoder

cat_enc = data[['City']]

print(cat_enc['City'].unique())

le = LabelEncoder()
cat_enc_le = le.fit_transform(cat_enc['City'])

print(le.classes_)
print(cat_enc_le)
print(np.unique(cat_enc_le))

#Мы получаем набор городов в цифровом формате. Они сортируются по алфавиту, поэтому последовательность мало, что даёт для понимания

#Теперь сделаем методом OrdinalEncoder

cat_oe = data[['City']]

oe = OrdinalEncoder()
cat_enc_oe = oe.fit_transform(cat_oe)

print(np.unique(cat_enc_oe))
