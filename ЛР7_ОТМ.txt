def test_stationarity(timeseries, title):
    """Проверка стационарности ряда с помощью теста Дики-Фуллера"""
    print(f'\nРезультаты теста для: {title}')
    dftest = adfuller(timeseries, autolag='AIC')
    dfoutput = pd.Series(dftest[0:4], index=['Test Statistic', 'p-value',
                                             '#Lags Used', 'Number of Observations Used'])
    for key, value in dftest[4].items():
        dfoutput['Critical Value (%s)' % key] = value

    print(dfoutput)

    if dftest[1] <= 0.05:
        print("✓ Ряд стационарен (p-value <= 0.05)")
        return True
    else:
        print("✗ Ряд нестационарен (p-value > 0.05)")
        return False

def evaluate_forecast(y_true, y_pred, model_name):
    """Расчет метрик качества прогноза"""
    mae = mean_absolute_error(y_true, y_pred)
    mse = mean_squared_error(y_true, y_pred)
    rmse = np.sqrt(mse)
    mape = np.mean(np.abs((y_true - y_pred) / y_true)) * 100

    print(f"\nМетрики для {model_name}:")
    print(f"  MAE:  {mae:.2f}")
    print(f"  MSE:  {mse:.2f}")
    print(f"  RMSE: {rmse:.2f}")
    print(f"  MAPE: {mape:.2f}%")

    return {'model': model_name, 'mae': mae, 'mse': mse, 'rmse': rmse, 'mape': mape}

def create_features_for_sr(series, lags=12):
    """Создание лаговых признаков для прогнозирования временного ряда"""
    df = pd.DataFrame()
    df['target'] = series.values

    # Добавление лагов
    for lag in range(1, lags + 1):
        df[f'lag_{lag}'] = df['target'].shift(lag)

    # Добавление временных признаков
    df['month'] = series.index.month
    df['year'] = series.index.year

    # Добавление скользящих статистик
    df['rolling_mean_3'] = df['target'].rolling(3).mean()
    df['rolling_std_3'] = df['target'].rolling(3).std()
    df['rolling_mean_6'] = df['target'].rolling(6).mean()
    df['rolling_mean_12'] = df['target'].rolling(12).mean()

    return df.dropna()
